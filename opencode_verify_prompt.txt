<S-001 STATUS="active">
      <Name>zenmoney-balance</Name>
      <Purpose>
        Fetch current account balances from ZenMoney API (read-only) and print summary to stdout.
        Not in scope: creating/updating/deleting any entities; OAuth2 interactive flows; long-running sync.
      </Purpose>

      <Interface>
        <Entrypoint>~/.local/bin/zenmoney-balance</Entrypoint>
        <Schedule>manual (optional cron later)</Schedule>
        <Args>
          --account &lt;id&gt; (optional; filter to one account)
          archived accounts are INCLUDED by default
          --exclude-archived (exclude archived accounts)
          --in-balance-only (default: true)
          --all-accounts (set inBalance-only to false)
          --timeout-sec &lt;n&gt; (default: 20)
          --full (expanded output: account list + per-currency totals + total_base)
          --json (stdout JSON-only; logs go to stderr)
        </Args>
        <EnvVars>
          ZENMONEY_API_KEY (provided by user; used as Bearer token value)
        </EnvVars>
        <Config>none</Config>
        <Outputs>
          default (no flags):
            narrative logs ending with:
              OK — balance fetched
            and includes:
              Info — total_base[BASE_CUR]=...

          --full:
            one line per account:
              &lt;title&gt; [&lt;CUR&gt;] (&lt;uuid&gt;)  &lt;balance&gt; (archived)   # suffix only when archive=true
            plus totals per currency:
              Info — total[CUR]=...
            plus total converted into user's base currency (user.currency):
              Info — total_base[BASE_CUR]=...
            final line:
              OK — balance fetched

          --json:
            stdout JSON-only (no narrative logs on stdout)
        </Outputs>
        <DryRun>not applicable (read-only)</DryRun>
      </Interface>

      <Language>
        <Proposed>Python 3 (stdlib-only HTTP client)</Proposed>
        <Alternative>Node.js 18+ (built-in fetch)</Alternative>
        <Rationale>
          Both can be dependency-free. Python tends to be simplest for cron + small CLIs; Node is fine if you prefer JS ecosystem.
        </Rationale>
        <Decision STATUS="approved">Python 3</Decision>
      </Language>

      <Dependencies>
        <Runtime>Python 3.x</Runtime>
        <ExternalTools />
        <ThirdPartyLibraries />
      </Dependencies>

      <Pipeline>
        <step-1>Validate args and required env var (ZENMONEY_API_KEY present; timeout numeric).</step-1>
        <step-2>Call ZenMoney API Diff endpoint (POST https://api.zenmoney.ru/v8/diff/) with Authorization: Bearer &lt;ZENMONEY_API_KEY&gt;.</step-2>

        <step-6>Print per-account balances + totals per currency + total_base converted into base currency; exit 0.</step-6>
